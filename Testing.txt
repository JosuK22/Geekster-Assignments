import java.io.*;
import java.util.*;
class Solution {
    static int m = 4;
    static int n = 4;
    static void row_sum(int arr[][])
    {

        int i, j, sum = 0;
        int row = 0;
        for (i = 0; i < m; ++i) {
            for (j = 0; j < n; ++j) {

            
                sum = sum + arr[i][j];
            }
            
            if(sum > row){
                row = sum;    
            }
           
            }
            sum = 0;
         System.out.println("row" + "0" + " "+ row );
        }
        
    
    static void column_sum(int arr[][])
    {

        int i, j, sum = 0;
        int  col = 0;
        for (i = 0; i < m; ++i) {
            for (j = 0; j < n; ++j) {
                sum = sum + arr[j][i];
            }
            if(sum > col){
                col = sum;
            }
            sum = 0;
        }
        System.out.println("col" + "0" + " "+  col );
    }
    static void maximum(){
        
    }

  
    public static void main(String[] args)
    {
        int i, j;
        int[][] arr = new int[m][n];

        int x = 1;
        for (i = 0; i < m; i++)
            for (j = 0; j < n; j++)
                arr[i][j] = x++;

    
        if(m==2){
        row_sum(arr);}
        else{
        column_sum(arr);}
        
        
    }
}
